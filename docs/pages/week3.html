df1 = pd.DataFrame({

Apply Functions
# Apply function to column
df['salary_category'] = df['salary'].apply(
    lambda x: 'High' if x > 60000 else 'Medium' if x > 50000 else 'Low'
)

# Apply custom function
def calculate_bonus(row):
    if row['department'] == 'IT':
        return row['salary'] * 0.15
    else:
        return row['salary'] * 0.10

df['bonus'] = df.apply(calculate_bonus, axis=1)

# Transform
df['salary_normalized'] = df.groupby('department')['salary'].transform(
    lambda x: (x - x.mean()) / x.std()
)

print(df)

Python Code Examples

Statistical Analysis
import numpy as np
import pandas as pd
from scipy import stats

# Sample data
np.random.seed(42)
data = np.random.normal(100, 15, 1000)  # mean=100, std=15

# Central tendency
mean = np.mean(data)
median = np.median(data)
mode = stats.mode(data, keepdims=True)[0][0]

# Dispersion
variance = np.var(data)
std_dev = np.std(data)
range_val = np.ptp(data)  # peak to peak
iqr = stats.iqr(data)  # Interquartile range

# Percentiles
q1 = np.percentile(data, 25)
q3 = np.percentile(data, 75)

print(f"Mean: {mean:.2f}")
print(f"Median: {median:.2f}")
print(f"Standard Deviation: {std_dev:.2f}")
print(f"IQR: {iqr:.2f}")
Correlation Analysis
# Create sample dataset
df_stats = pd.DataFrame({
    'hours_studied': np.random.randint(1, 10, 100),
    'practice_tests': np.random.randint(0, 20, 100),
    'score': np.random.randint(50, 100, 100)
})

# Correlation matrix
correlation = df_stats.corr()
print("Correlation Matrix:")
print(correlation)

# Specific correlation
corr_study_score = df_stats['hours_studied'].corr(df_stats['score'])
print(f"\nCorrelation between study hours and score: {corr_study_score:.3f}")

# Covariance
covariance = df_stats.cov()
print("\nCovariance Matrix:")
print(covariance)

Sales Analysis Project
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# Generate sample sales data
np.random.seed(42)
dates = pd.date_range('2023-01-01', periods=365, freq='D')
sales_data = pd.DataFrame({
    'date': dates,
    'product': np.random.choice(['Product A', 'Product B', 'Product C'], 365),
    'region': np.random.choice(['North', 'South', 'East', 'West'], 365),
    'sales': np.random.randint(100, 1000, 365),
    'quantity': np.random.randint(1, 50, 365),
    'discount': np.random.choice([0, 5, 10, 15], 365)
})

# Data cleaning
sales_data['revenue'] = sales_data['sales'] * (1 - sales_data['discount']/100)
sales_data['month'] = sales_data['date'].dt.month
sales_data['quarter'] = sales_data['date'].dt.quarter

# Analysis 1: Monthly sales trend
monthly_sales = sales_data.groupby('month')['revenue'].sum()

plt.figure(figsize=(12, 6))
plt.plot(monthly_sales.index, monthly_sales.values, marker='o')
plt.xlabel('Month')
plt.ylabel('Revenue')
plt.title('Monthly Revenue Trend')
plt.grid(True)
plt.show()

# Analysis 2: Product performance
product_analysis = sales_data.groupby('product').agg({
    'revenue': 'sum',
    'quantity': 'sum',
    'sales': 'mean'
}).round(2)

print("Product Performance:")
print(product_analysis)

# Analysis 3: Regional comparison
plt.figure(figsize=(10, 6))
sns.boxplot(x='region', y='revenue', data=sales_data)
plt.title('Revenue Distribution by Region')
plt.show()

# Analysis 4: Correlation analysis
plt.figure(figsize=(8, 6))
corr = sales_data[['sales', 'quantity', 'discount', 'revenue']].corr()
sns.heatmap(corr, annot=True, cmap='coolwarm')
plt.title('Correlation Matrix')
plt.show()

# Generate summary report
print("\n=== SALES ANALYSIS REPORT ===")
print(f"Total Revenue: ${sales_data['revenue'].sum():,.2f}")
print(f"Average Daily Sales: ${sales_data['revenue'].mean():,.2f}")
print(f"Best Performing Product: {product_analysis['revenue'].idxmax()}")
print(f"Best Performing Region: {sales_data.groupby('region')['revenue'].sum().idxmax()}")
