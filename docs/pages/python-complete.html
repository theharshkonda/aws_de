Introduction to Python

Python is a high-level, interpreted programming language known for its simplicity and readability. It's the most popular language for data engineering due to its extensive libraries and ease of use.

Why Python for Data Engineering?

  • Easy to Learn: Clean syntax and readability
  • Rich Ecosystem: Pandas, NumPy, PySpark, etc.
  • Integration: Works with all major data platforms
  • Community: Huge community and support

4. Sets (Unordered, Unique)

Working with Sets
# Creating sets
numbers = {1, 2, 3, 4, 5}
duplicates = {1, 2, 2, 3, 3, 3}  # {1, 2, 3}

# Set operations
numbers.add(6)          # Add element
numbers.remove(1)       # Remove (error if not found)
numbers.discard(10)     # Remove (no error)

# Set mathematics
a = {1, 2, 3, 4}
b = {3, 4, 5, 6}

print(a | b)    # {1, 2, 3, 4, 5, 6} (union)
print(a & b)    # {3, 4} (intersection)
print(a - b)    # {1, 2} (difference)
print(a ^ b)    # {1, 2, 5, 6} (symmetric difference)

# Use cases in Data Engineering
# 1. Remove duplicates
data = [1, 2, 2, 3, 3, 3, 4]
unique = list(set(data))  # [1, 2, 3, 4]

# 2. Find unique values
ids = {100, 101, 102, 103}
new_ids = {102, 103, 104, 105}
truly_new = new_ids - ids  # {104, 105}
Data Structure Choice Guide:
  • List: Ordered collection, allow duplicates, need to modify
  • Tuple: Ordered collection, immutable, function returns
  • Dictionary: Key-value pairs, fast lookups, JSON-like data
  • Set: Unique elements, mathematical operations, deduplication

Function Basics

2. Advanced Functions

*args and **kwargs
# *args - variable number of arguments
def sum_all(*args):
    return sum(args)

print(sum_all(1, 2, 3))        # 6
print(sum_all(1, 2, 3, 4, 5))  # 15

# **kwargs - variable keyword arguments
def create_profile(**kwargs):
    return kwargs

profile = create_profile(name="Alice", age=28, city="NYC")
# {'name': 'Alice', 'age': 28, 'city': 'NYC'}

# Combining all
def func(a, b, *args, **kwargs):
    print(f"a: {a}, b: {b}")
    print(f"args: {args}")
    print(f"kwargs: {kwargs}")

func(1, 2, 3, 4, 5, x=10, y=20)

# Lambda functions (anonymous functions)
square = lambda x: x ** 2
print(square(5))  # 25

add = lambda a, b: a + b
print(add(3, 4))  # 7

# Lambda with map, filter, reduce
numbers = [1, 2, 3, 4, 5]

# map - apply function to each element
squared = list(map(lambda x: x**2, numbers))
# [1, 4, 9, 16, 25]

# filter - keep elements that match condition
evens = list(filter(lambda x: x % 2 == 0, numbers))
# [2, 4]

# reduce - reduce to single value
from functools import reduce
total = reduce(lambda a, b: a + b, numbers)
# 15

3. Data Engineering Functions

Practical Data Engineering Functions
import json
from datetime import datetime

def validate_record(record, required_fields):
    """
    Validate if record has all required fields
    """
    missing_fields = []
    for field in required_fields:
        if field not in record or record[field] is None:
            missing_fields.append(field)

    return len(missing_fields) == 0, missing_fields

def clean_data(data):
    """
    Clean and standardize data
    """
    cleaned = []
    for record in data:
        # Remove null values
        if record.get('value') is not None:
            # Standardize date format
            if 'date' in record:
                record['date'] = datetime.strptime(
                    record['date'], '%Y-%m-%d'
                ).isoformat()
            cleaned.append(record)
    return cleaned

def transform_records(records, mapping):
    """
    Transform records using field mapping
    """
    transformed = []
    for record in records:
        new_record = {}
        for old_field, new_field in mapping.items():
            if old_field in record:
                new_record[new_field] = record[old_field]
        transformed.append(new_record)
    return transformed

# Usage example
raw_data = [
    {"user_id": 1, "user_name": "Alice", "amount": 100},
    {"user_id": 2, "user_name": "Bob", "amount": None},
    {"user_id": 3, "user_name": "Charlie", "amount": 300}
]

# Validate
required = ["user_id", "user_name", "amount"]
for record in raw_data:
    is_valid, missing = validate_record(record, required)
    if not is_valid:
        print(f"Invalid record: missing {missing}")

# Transform
mapping = {
    "user_id": "id",
    "user_name": "name",
    "amount": "total"
}
transformed = transform_records(raw_data, mapping)
